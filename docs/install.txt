// firebase 설정

// firebase deploy
firebase deploy --only functions

// 특정 유저 구매만 삭제
firebase firestore:delete users/<UID>/purchases --recursive -f --project devian-framwork-example

// 해당 유저 entitlement 문서도 제거(또는 콘솔에서 수동 초기화)
firebase firestore:delete users/<UID>/entitlements/current -f --project devian-framwork-example

// DB 전체 삭제(진짜 전체 초기화, 매우 위험)
firebase firestore:delete --all-collections -f --project devian-framwork-example

// 보안 해제
xattr -dr com.apple.quarantine Assets/Firebase
xattr -dr com.apple.quarantine Assets/Plugins

// Generate keystore
keytool -genkeypair -v \
  -keystore {filename}.keystore \
  -storepass {passwd} -keypass {passwd} \
  -alias example \
  -keyalg RSA -keysize 2048 -validity 10000 \
  -dname "CN=Example,O=Debug,C=KR"

// SHA-1
keytool -exportcert -rfc \
  -alias example \
  -keystore {filename}.keystore \
  -storepass {passwd} -keypass {passwd} \
| openssl x509 -noout -fingerprint -sha1
